#!/usr/bin/env python3
"""Project-local Agent tuning test runner.

Generated by the agent-tuning-generate-runner Skill.
- Preserves all original dataset columns and order.
- Appends required actual-output column: agent_output.
- Writes results to agent-tuning/results/vN/results.csv.
- Optionally writes agent-tuning/results/vN/app.log when log capture is configured.
"""

from __future__ import annotations

import csv
import json
import sys
import traceback
from contextlib import redirect_stderr, redirect_stdout
from pathlib import Path
from typing import Any, Dict, Iterable, List, Tuple

RESULTS_DIR = Path("agent-tuning/results")
RUNNER_DIR = Path("agent-tuning/runner")
DATASET_PATH = Path("TODO_AGENT_TUNING_DATASET_PATH")
RESULTS_FILENAME = "results.csv"
APP_LOG_FILENAME = "app.log"


class UserActionRequired(RuntimeError):
    """Raised when the user must repair inputs or confirm an unsafe inference."""


class AgentExecutionError(RuntimeError):
    """Raised for a known per-row Agent execution failure that should be recorded."""


def list_version_dirs(results_dir=RESULTS_DIR):
    # Return [(number, path)] for directories named vN where N is a positive integer.
    # Missing results_dir means no versions exist yet; runner first run will create v1.
    if not results_dir.exists():
        return []
    return sorted(
        (int(path.name[1:]), path)
        for path in results_dir.iterdir()
        if path.is_dir() and path.name.startswith("v") and path.name[1:].isdigit() and int(path.name[1:]) > 0
    )


def allocate_next_results_version(results_dir=RESULTS_DIR):
    # Used only by test_runner.py. Current max version determines create vs reuse.
    versions = list_version_dirs(results_dir)
    if not versions:
        target = results_dir / "v1"
    else:
        max_n, current = versions[-1]
        target = results_dir / f"v{max_n + 1}" if (current / "results.csv").exists() else current
    target.mkdir(parents=True, exist_ok=True)
    return target


def load_dataset(path: Path) -> Tuple[List[str], List[Dict[str, str]]]:
    """Load CSV rows while preserving original columns.

    TODO_AGENT_TUNING: adapt this function only if the dataset is not CSV.
    Do not rename user columns here. If the source dataset contains agent_output,
    the Skill must confirm a rename strategy before generating this script.
    """
    if not path.exists():
        raise UserActionRequired(f"Dataset not found: {path}")
    with path.open("r", newline="", encoding="utf-8-sig") as handle:
        reader = csv.DictReader(handle)
        if not reader.fieldnames:
            raise UserActionRequired(f"Dataset has no header row: {path}")
        fieldnames = list(reader.fieldnames)
        if "agent_output" in fieldnames:
            raise UserActionRequired(
                "Dataset already contains agent_output; confirm a rename strategy before running."
            )
        return fieldnames, list(reader)


def build_agent_input(row: Dict[str, str]) -> Any:
    """Convert a dataset row into the local Agent input.

    TODO_AGENT_TUNING: fill this after inspecting Agent source and dataset columns.
    Example patterns:
    - return row["question"]
    - return {"messages": [{"role": "user", "content": row["query"]}]}
    - return {"input": row["input"], "context": row.get("context", "")}
    """
    raise UserActionRequired("TODO_AGENT_TUNING: implement build_agent_input(row).")


def call_agent(agent_input: Any) -> Any:
    """Call the local Agent and return its raw output.

    TODO_AGENT_TUNING: import and call the target Agent here. Prefer the target
    project's own virtualenv/dependencies. Handle async explicitly if needed.
    Raise AgentExecutionError only for known per-row failures that should be
    preserved in results.csv; let configuration/import errors fail the run.
    """
    raise UserActionRequired("TODO_AGENT_TUNING: implement call_agent(agent_input).")


def normalize_agent_output(raw_output: Any) -> str:
    """Serialize Agent output for the required agent_output column."""
    if isinstance(raw_output, str):
        return raw_output
    return json.dumps(raw_output, ensure_ascii=False, sort_keys=True)


def iter_result_rows(rows: Iterable[Dict[str, str]]) -> Iterable[Dict[str, str]]:
    for index, row in enumerate(rows, start=1):
        output_row = dict(row)
        try:
            agent_input = build_agent_input(row)
            raw_output = call_agent(agent_input)
            output_row["agent_output"] = normalize_agent_output(raw_output)
            output_row["agent_output_status"] = "ok"
        except UserActionRequired:
            # Configuration/TODO/confirmation failures must stop the run; otherwise
            # an invalid runner could create a misleading successful results.csv.
            raise
        except AgentExecutionError as exc:
            # Known per-row Agent failures are still useful evaluation evidence.
            output_row["agent_output"] = json.dumps(
                {
                    "error": type(exc).__name__,
                    "message": str(exc),
                    "row_number": index,
                    "traceback": traceback.format_exc(limit=8),
                },
                ensure_ascii=False,
            )
            output_row["agent_output_status"] = "error"
            output_row["agent_output_error_type"] = type(exc).__name__
            output_row["agent_output_error_message"] = str(exc)
        yield output_row


def write_results(version_dir: Path, source_fieldnames: List[str], rows: List[Dict[str, str]]) -> Path:
    output_path = version_dir / RESULTS_FILENAME
    auxiliary_fields = sorted(
        {key for row in rows for key in row if key.startswith("agent_output_") and key not in source_fieldnames}
    )
    fieldnames = list(source_fieldnames) + ["agent_output"] + auxiliary_fields
    with output_path.open("w", newline="", encoding="utf-8") as handle:
        writer = csv.DictWriter(handle, fieldnames=fieldnames, extrasaction="ignore")
        writer.writeheader()
        writer.writerows(rows)
    return output_path


def run() -> int:
    version_dir = allocate_next_results_version()
    source_fieldnames, dataset_rows = load_dataset(DATASET_PATH)
    result_rows = list(iter_result_rows(dataset_rows))
    output_path = write_results(version_dir, source_fieldnames, result_rows)
    print(f"Wrote {len(result_rows)} rows to {output_path}")
    return 0


if __name__ == "__main__":
    # TODO_AGENT_TUNING_LOG_CAPTURE:

    # If no reliable log source exists, call sys.exit(run()) directly and omit app.log.
    target_version_dir = allocate_next_results_version()
    log_path = target_version_dir / APP_LOG_FILENAME

    # Reuse the preallocated version directory for this process so log and results align.
    def allocate_next_results_version(results_dir=RESULTS_DIR):  # type: ignore[no-redef]
        target_version_dir.mkdir(parents=True, exist_ok=True)
        return target_version_dir

    with log_path.open("w", encoding="utf-8") as log_handle:
        with redirect_stdout(log_handle), redirect_stderr(log_handle):
            exit_code = run()
    print(f"Wrote optional app log to {log_path}")
    sys.exit(exit_code)
